\begin{frame}
 \frametitle{Analiza}

Iskorišæenost servera: $\rho = \lambda / \mu$\\[1ex]
Pollaczek-Khinchinova formula
 \begin{columns}[T]
	\column{.35\textwidth}
	 \begin{block}{$M$/$G$/1}<1->
	$$T_{Q}=\frac{\lambda \overline{\tau^{2}}}{2(1-\rho)}
$$
\end{block}
	\column{.55\textwidth}
  \begin{exampleblock}{$M$/$D$/1}<2->
	$$\overline{\tau^{2}}=\frac{1}{\mu^{2}} \quad \Rightarrow \quad
	T_{Q}=\frac{\rho}{2\mu(1-\rho)}
$$
\end{exampleblock}
\end{columns}
\end{frame}
